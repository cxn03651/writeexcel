153 
154       # Skip this if another method handles the continue blocks.
155       return data if @ignore_continue != 0
156

158       # the length field of the record.

163       if data.length > @limit
164          tmp = data[0, @limit]
165          data[0, @limit] = ''
166       else
167          tmp = data.dup
168          data = ''
169       end

174       while (data.length > @limit)
175          header  = [record, @limit].pack("vv")
